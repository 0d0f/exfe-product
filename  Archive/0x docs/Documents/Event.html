Current invitation or invitation for current user, User.current. From the view of current user, invitation can be treat as a nested singelton resource of events.

So we can route Invitation RSVP like this:

get '/:event_id/yes' => 'invitations#respond_yes', :as => :respond_yes

Notice InvitationsController is belongs_to EventsController as singelton.

def self.find_by_base62_id_or_code(base62_id_or_code)
  if base62_id_or_code =~ /\A!(.*)\Z/
    find($1.as_base62_to_i)
  else
    base62_id = base62_id_or_code.as_base62_to_i
    find_by_code(base62_id_or_code) or find(base62_id)
  end
end

def self.for_user(user)
  joins(:invitations).
    where(
      {:creator_id => user.id} |
      {:invitations => {:invited_identity_id => user.identity_ids}}
    ).
    group('events.id')
end

def begin_at_date=(begin_at_date)
  return if begin_at_date.blank?

  @begin_at_date = begin_at_date
  date = Date.parse(begin_at_date)

  self.begin_at ||= Time.now
  self.begin_at = (date - self.begin_at.to_date).days.since self.begin_at
  @begin_at_date_invalid = false
rescue ArgumentError
  @begin_at_date_invalid = true
end

def begin_at_time=(begin_at_time)
  return if begin_at_time.blank?

  @begin_at_time = begin_at_time
  time = Time.parse(begin_at_time)

  self.begin_at ||= Time.now
  self.begin_at = self.begin_at.midnight.since time.seconds_since_midnight

  @begin_at_time_invalid = false
rescue ArgumentError
  @begin_at_time_invalid = true
end

def invited?(user_or_identity)
  case user_or_identity

  when User
    !(invited_identities & user_or_identity.identities).empty?

  when Identity
    invited_identities.include?(user_or_identity)

  end
end
