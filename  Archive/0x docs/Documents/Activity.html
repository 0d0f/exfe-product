def message
  message_target = target
  message_data = data

  case data
  when Invitation
    message_data = target
  when Comment
    message_target = data 
    message_extra = target
  end

  I18n.t(
    action,
    :scope => [:activities, target.class.model_name.i18n_key],
    :target => message_target.class.model_name.human,
    :data => message_data,
    :extra=> message_extra
  )
end
