def initialize(user_or_invitation = nil)
  user_or_invitation ||= User.new # guest user (not logged in)

  case user_or_invitation
  when Admin
    can :manage, :all
  when User
    define_user_permissions(user_or_invitation)
  when Invitation
    define_invitation_permissions(user_or_invitation)
  end
end

def define_invitation_permissions(invitation)

  can :manage, Invitation, :id => invitation.id, :state => 'pending'
  can :read, Invitation, :id => invitation.id
end

def define_user_permissions(user)
  can :manage, Event, :creator_id => user.id
  can :read, Event, :state => 'published', :invitations => {:invited_identity_id => user.identity_ids}
  can :create, Event unless user.new_record?

  can :manage, Invitation, :invited_identity_id => user.identity_ids

  can :manage, User, :id => user.id
  can :read, User unless user.new_record?

  can [:create, :read], Identity, :user_id => user.id

  can :manage, Comment do |comment|
    user.identity_ids.include?(comment.identity_id) ||
      comment.commentable.creator_id == user.id
  end

  can [:create, :read], Comment do |comment|
    if comment.commentable.is_a?(Event)
      comment.commentable.invitations.any? do |invitation|
        user.identity_ids.include?(invitation.invited_identity_id)
      end
    end
  end

  can :read, Activity unless user.new_record?
end
