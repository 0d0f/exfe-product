def as_base62_to_i
  raise ArgumentError, 'Invalid base-62 number.' unless self.is_valid_base62?
  value = 0
  position = 0
  self.reverse.each_char do |c|
    value += 62 ** position * Base62Values[c]
    position += 1
  end
  value
end
