def self.accessible_by_user(user)
  joins(:commentable.type(Event) => :invitations).
    where(
      {:commentable => {:creator_id => user.id}} |
      {:comments => {:identity_id => user.identity_ids}} |
      {:commentable => {:invitations => {:invited_identity_id => user.identity_ids}}}
    ).
    group('comments.id')
end
