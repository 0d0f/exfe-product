def create
  params[:comment] = clean_params
  params[:comment].delete :event_id

  resource.identity = current_identity

  if resource.update_attributes(params[:comment])
    render :json => resource
  else
    render :json => {:error => resource.errors}
  end
end

def index
  authorize! :index, Comment

  case parent_type
  when :user
    comments_scope = Comment.for_user(@user)
  else
    comments_scope = parent.comments
  end

  @comments = apply_scopes(comments_scope).accessible_by_user(current_user)

  index!
end
