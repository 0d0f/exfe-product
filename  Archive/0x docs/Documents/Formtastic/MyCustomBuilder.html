def datetime_input(method, options)
  date_method = "#{method}_date"
  time_method = "#{method}_time"

  html_options = options.delete(:input_html) || {}
  date_html_options = default_string_options(date_method, :string).merge(html_options)
  time_html_options = default_string_options(time_method, :string).merge(html_options)

  date_field_id = generate_html_id(date_method, "")
  time_field_id = generate_html_id(time_method, "")

  date_html_options[:id] ||= date_field_id
  time_html_options[:id] ||= time_field_id
  label_options = options_for_label(options)
  label_options[:for] ||= date_html_options[:id]

  label(method, label_options) <<
    send(:text_field, date_method, date_html_options) <<
    send(:text_field, time_method, time_html_options)
end

def identity_text_input(method, options)
  invited_identity_list_as_string = if @object.send(method).respond_to?(:join)
                                      @object.send(method).join(' ')
                                    else
                                      @object.send(method)
                                    end
  label_options = options_for_label(options)
  label_options[:for] ||= options[:id]
  label(method, label_options) <<
    template.text_area_tag("#{object_name}[#{method}]", invited_identity_list_as_string, options)
end
