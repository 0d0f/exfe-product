def method_missing(method, *args, &block)
  # Also pass _methods to avoid error introduced by this commit in Rails 3.0.6
  # http://goo.gl/GDXPJ
  if respond_to?(method) || method.to_s[0, 1] == '_'
    super
  elsif extra
    extra.send(method, *args, &block)
  end
end

def self.create_or_update(params)
    token=where("auth_token = ?", params[:token])
    if token.exists?
      identity = token.first
      if identity.user_id == User.current.id
        return identity
      end
    else
      identity= Identity.new
     end
     identity.user_id= User.current.id
     identity.username= User.current.name
     identity.auth_token=params[:token]
     identity.provider=params[:provider]
     identity.save!
end

def self.iosapn_token(uid)
    token=where("user_id= ? and provider = ?", uid,"iOSAPN")
    if token.exists?
      identity = token.first
      return identity.auth_token
    end
    return ""
end

def serializable_hash(options = nil)

  unless user == User.current

    options[:except] += PRIVATE_ATTRIBUTES
  end

  super
end
