def serializable_hash(options = nil)
  options ||= {}
  options = options.merge(:except => ATTRIBUTES_FOR_DEVISE.dup)
  options[:only] << 'id' if options[:only]
  if self == User.current
    options[:except].delete 'authentication_token'
    # Devise 1.3.4 use :only option in #to_json to limit to accessible attributes
    options[:only] << 'authentication_token' if options[:only]
  end
  super
end
