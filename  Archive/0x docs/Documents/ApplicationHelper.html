def link_to_identity(identity, options = {})
  if options[:avatar]
    link_to(image_tag(identity.avatar.url(options[:avatar])), identity.user, :title => identity)
  else
    link_to(identity, identity.user, :title => identity)
  end
end

def link_to_user(user, options = {})

    else
      link_to(user, user, :title => user)
  end
end
